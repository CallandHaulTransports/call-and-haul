    const bookingStatus = document.getElementById('bookingStatus');
    bookingForm.addEventListener('submit', function(e){
      e.preventDefault();
      const data = new FormData(bookingForm);
      // For a static site, you can POST to Formspree or use fetch to your backend.
      // Here, we fake a successful send and show instructions.
      bookingStatus.textContent = 'Sending...';

        bookingStatus.textContent = 'Request sent — we will contact you within business hours.';

        setTimeout(closeModal, 1500);
      }, 800);
    });

    const quickQuote = document.getElementById('quickQuote');

      e.preventDefault();

      setTimeout(()=> {
        statusEl.textContent = 'Quote request received — check your email or phone shortly.';
        quickQuote.reset();
